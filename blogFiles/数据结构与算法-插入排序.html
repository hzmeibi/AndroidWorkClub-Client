概念

每一步将一个待排序的元素，按其大小插入已经排序的序列中的适当位置，直到全部插入完成。适用于数据量较小的排序。

实现

insertSort

    /**
     * 插入排序
     *
     * @param array
     */
    public static void insertSort(int[] array) {
        // 第1个数肯定是有序的，从第2个数开始遍历，依次插入有序序列
        for (int i = 1; i < array.length; i++) {
            int temp = array[i]; // 取出第i个数，
            int insertPoint = i - 1;//
            //循环比对前i-1个数和当前要插入temp的大小，插入合适位置
            //从已经排序的序列中，找到第一个小于或等于temp的位置将temp插入，
            while (insertPoint >= 0 && array[insertPoint] > temp) {
                //循环的过程中需要预留temp的空间,所以每次循环需要将元素后移一个位置
                array[insertPoint + 1] = array[insertPoint];
                insertPoint--;
            }
            //当前位置值小于或等于temp，交换值
            array[insertPoint + 1] = temp;
        }
    }

main

    public static void main(String[] arg) {
        int[] array = new int[]{6, 1, 4, 2, 3, 5, 10};
        System.out.println("排序之前：");
        for (int element : array) {
            System.out.print(element + " ");
        }
        insertSort(array);
        System.out.println("\n排序之后：");
        for (int element : array) {
            System.out.print(element + " ");
        }
    }

输出结果：

排序之前：
6 1 4 2 3 5 10 
排序之后：
1 2 3 4 5 6 10
